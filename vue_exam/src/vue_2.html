조건문

        <div id="app-3" class="text">
            <p v-if="seen">보임?</p>
            
            var app3 = new Vue( { 
                el : "#app-3",
                data : {
                    seen:true
                }
            })

        TEST
        console> app3.seen = false

        <div id="app-for" class="text">
            <ul>
                <li v-for="todo in todos">
                    {{todo.no}}. {{todo.text}}
                </li>
            </ul>
        </div>

        var app_for = new Vue( { 
            el : "#app-for",
            data : {
                todos : [
                    {no:1, text:'first'},
                    {no:2, text:'second'},
                    {no:3, text:'third'}
                ]
            }
        })

        TEST
        console>app_for.todos.push({no:4, text:'fourth'})
